46static inline std::vector<std::pair<Node, Node>> as_nodepairs(
47 const std::vector<std::pair<unsigned, unsigned>> &edges) {
48 std::vector<std::pair<Node, Node>> nodepairs;
49 nodepairs.reserve(edges.size());
50 for (auto &[m, n] : edges) {
51 nodepairs.push_back({Node(m), Node(n)});
52 }
53 return nodepairs;
54}

175// Subclass, constructor generates adjacency matrix corresponding to a
176// SquareGrid architecture
177class SquareGrid : public Architecture {
178 public:
179 // Converts square indexing to qubit indexing
180 Vertex squind_to_qind(
181 const unsigned ver, const unsigned hor, const unsigned layer = 0) const {
182 return (ver * dimension_c + hor) + single_layer_nodes() * layer;
183 }
184 // Returns number of nodes in a single 2d layer
185 unsigned single_layer_nodes() const { return dimension_c * dimension_r; }
186 // Gets number of columns of square grid architecture
187 unsigned get_columns() const { return dimension_c; }
188 // Gets number of rows of square grid architecture
189 unsigned get_rows() const { return dimension_r; }
190 // Gets number of layers of square grid architecture
191 unsigned get_layers() const { return layers; }
192 // Converts qubit indexing to square indexing
193 std::pair<unsigned, unsigned> qind_to_squind(const Vertex &qn) const {
194 unsigned col = qn % dimension_c;
195 unsigned row = (qn - col) / dimension_c;
196 return {row, col};
197 }
198 // SquareGrid constructor
199 // dim_c equiv 'x', dim_r equiv 'y'
200 SquareGrid(
201 const unsigned dim_r, const unsigned dim_c, const unsigned _layers = 1,
202 const std::string &label = "gridNode");
203
204 private:
205 static std::vector<Connection> get_edges(
206 const unsigned dim_r, const unsigned dim_c, const unsigned layers = 1,
207 const std::string &label = "gridNode");
208
209 unsigned dimension_r;
210 unsigned dimension_c;
211 unsigned layers;
212};
